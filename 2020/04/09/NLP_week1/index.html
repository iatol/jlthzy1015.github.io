def sta_word(filename):
    my_dict={}
    text=loadData(filename)
    for char in text:
        #在，频率就+1
        if char in my_dict:
            my_dict[char] = my_dict[char]+1
        #不在，就创建一个位置给它
        else:
            my_dict[char] = 1

    return my_dict
